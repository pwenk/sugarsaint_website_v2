### 2.4 Article Generation Guidelines

**Tone & Style:**

**For B2B Audience (Corporations):**
- Use "Sie" (formal you)
- Professional, data-driven tone
- Emphasize ROI, risk mitigation, compliance
- Include business statistics
- Focus on preventive measures

**For B2C Audience (Individuals):**
- Use "Sie" (formal, but warmer)
- Empathetic, understanding tone
- Address emotional concerns (fear, privacy, trust)
- Include reassurance about discretion
- Focus on solving personal problems

**Universal Style Rules:**
- Clear, direct German (avoid jargon)
- Short paragraphs (3-4 sentences max)
- Active voice
- Professional but approachable
- Legal compliance (GDPR/DSGVO aware)

**SEO Optimization:**
- Use primary keyword naturally 3-5 times
- Include keyword in: H1, first paragraph, one H2, conclusion
- Use semantic variations
- Question-based H2 headings (for featured snippets)
- FAQ section (for FAQPage schema)

**Length:**
- Target: 1,000-1,500 words per article
- Minimum: 800 words
- Maximum: 2,000 words

## ARTICLE GENERATION SCRIPT

### Automated Generation Process

**Create Node.js script: `scripts/generate-articles.js`**

```javascript
const fs = require('fs');
const path = require('path');
const csv = require('csv-parser');

// Configuration
const CSV_PATH = '/Users/patrick/Library/CloudStorage/Dropbox/Mac/Documents/vibe/sugarsaint_website_v2/Bona Fides Blog Ideas.csv';
const OUTPUT_DIR = 'blog/articles';

// Parse CSV
function parseCSV() {
  return new Promise((resolve, reject) => {
    const articles = [];
    fs.createReadStream(CSV_PATH)
      .pipe(csv())
      .on('data', (row) => articles.push(row))
      .on('end', () => resolve(articles))
      .on('error', reject);
  });
}

// Generate slug from German title
function generateSlug(title) {
  return title
    .toLowerCase()
    .normalize('NFD')
    .replace(/[\u0300-\u036f]/g, '') // Remove diacritics
    .replace(/√§/g, 'ae')
    .replace(/√∂/g, 'oe')
    .replace(/√º/g, 'ue')
    .replace(/√ü/g, 'ss')
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '');
}

// Assign date based on priority
function assignDate(articleNum, priority) {
  const baseDate = new Date('2025-06-01');
  let offset = articleNum;

  if (priority === 'Priority 1') {
    offset += 120; // Recent dates
  } else if (priority === 'Priority 2') {
    offset += 60; // Mid dates
  }

  const date = new Date(baseDate);
  date.setDate(date.getDate() + offset);
  return date.toISOString().split('T')[0];
}

// Generate article content using AI
async function generateArticleContent(article) {
  // This is where you'd call Claude API or GPT-4 to generate content
  // For now, return template structure

  const slug = generateSlug(article['Article Title (German)']);
  const date = assignDate(parseInt(article['#']), article['Priority']);

  return `---
title: "${article['Article Title (German)']}"
slug: ${slug}
keyword: ${article['Primary Keyword']}
category: ${article['Category']}
date: ${date}
excerpt: "Professionelle Ermittlungshilfe f√ºr ${article['Category']} in Deutschland."
---

# ${article['Article Title (German)']}

## Zusammenfassung (TL;DR)

[AI-generated German summary based on category and title]

---

[AI-generated article content following the template]
`;
}

// Write article to file
function writeArticle(articleNum, content) {
  const filename = `${String(articleNum).padStart(3, '0')}-${generateSlug(content.title)}.md`;
  const filepath = path.join(OUTPUT_DIR, filename);
  fs.writeFileSync(filepath, content);
  console.log(`‚úÖ Generated: ${filename}`);
}

// Main execution
async function main() {
  console.log('üìñ Parsing CSV...');
  const articles = await parseCSV();

  console.log(`üìù Generating ${articles.length} articles...`);

  for (const article of articles) {
    const content = await generateArticleContent(article);
    writeArticle(parseInt(article['#']), content);
  }

  console.log('‚ú® All articles generated!');
}

main().catch(console.error);
```
